Grammer

Program -> Function Main

Fucntion -> FuncDef FuncBody Function | e

FuncDef -> ident -> Param Type

Param -> Type ident -> Param | e

Type -> int | char | string | bool | float

FuncBody -> Scope

Scope -> ScopeType { ScopeBody }

ScopeType -> 'Pure' | 'IO' | 'State' | 'Loop' | e

ScopeBody -> Code

Code -> Statement Code'| Loop Code'| Conditional Code'| Scope Code'

Code' -> Code | e

Statement -> IO | Return | Decloration

IO -> Print | Input | Output

Print -> print String ;

Input -> input ident Value ;

Output -> output String (Value) ( output ( String ) ;) ;

Return -> return Value (int) ;

Value -> String | Int | Char | Bool | Float | Expression

String -> string | idnet | StringConcat | Cast

Int -> int | ident

Char -> char | ident

Bool -> bool | ident

Float -> float | ident

Main -> MainDef FuncBody

MainDef -> main '->' 'int' (int main ( )) | int main ( int argc , string argv )

Decloration -> Mut | Let

Mut -> mut Type ident = Value (int) ;

Let -> Type ident = Value ;

Expression -> Term Expression'

Expression' -> Operator Term Expression' | e

Term -> Factor | ( Expression )

Factor -> Number

Number -> Int | Float

Operator -> + | - | * | / | %

Regex

int = ^-?\d+$

float = ^-?\d+\.\d+$

char = ^'[a-zA-Z0-9_]'

bool = \b(true|false)\b

string = ^\"([a-zA-Z0-9_]|\\[nt\"\\])*\"$

ident = ^[a-zA-Z_][a-zA-Z0-9_]*$
