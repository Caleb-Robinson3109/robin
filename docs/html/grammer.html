Program -> Function Main

Fucntion -> FuncDef FuncBody Function | e

FuncDef -> ident -> Param Type

Param -> Type ident -> Param | e

Type -> int | char | string | bool | float | type | ident | RetType | RetT

FuncBody -> Scope

Scope -> ScopeType { ScopeBody }

ScopeType -> 'Pure' | 'IO' | 'State' | 'Loop' | e

ScopeBody -> Code

Code -> Statement Code'| Loop Code'| Conditional Code'| Scope Code'

Code' -> Code | e

Statement -> IO | Return | Decloration | RetVoid

IO -> Print | Input | Output

Print -> print String ;

Input -> input ident Value ;

Output -> output String (Value) ( output ( String ) ;) ;

Return -> return Value (int) ;

Value -> String | Char | Bool | Expression | Type

String -> string | idnet | StringConcat | RetString | RetT

Int -> int | ident | RetInt | RetT

Char -> char | ident | RetChar | RetT

Bool -> bool | ident | RetBool | RetT

Float -> float | ident | RetFloat | RetT

Main -> MainDef FuncBody

MainDef -> int main ( )| int main ( int argc , string argv )

Decloration -> Mut | Let

Mut -> mut Type ident = Value (int) ;

Let -> Type ident = Value ;

Expression -> Term Expression'

Expression' -> Operator Term Expression' | e

Term -> Factor | ( Expression )

Factor -> Number

Number -> Int | Float

Operator -> + | - | * | / | %

Proposition -> Condition

Condition -> Unary Condition'

Condition' -> BooleanOperator Unary Condition' | e

Unary -> ! Unary | Primary

Primary -> Bool | ( Condition )

LogicalOperator -> || | && | ^^ | -> | == | !=

Branch -> BranchIf BranchElif BranchElse

Conditional -> If Elif Else

BranchIF -> if ( Proposition ) { Code }

BranchElif -> elif ( Proposition ) { Code } BranchElif | e

BranchElse -> else { Code } | e

If -> if ( Proposition ) { Value }

Elif -> elif (Proposition) {Value} Elif | e

Else -> else { Value }

TypeOf -> ( Value | Type ) ;

Cast -> cast ( Type , Value ) ;

KwFuncs -> TypeOf | Cast

Str -> str ( Value | Type )
